题目

细节题。打 +++ 调了 101010 分钟过了所有样例和我造的样例。

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int T;
ll n,k;
ll ans;
int main() {
    freopen("stairs.in","r",stdin);
    freopen("stairs.out","w",stdout);
    scanf("%d",&T);
    while(T--) {
        scanf("%lld %lld",&n,&k);
        if (n<k) printf("1\n");
        else {
            if (n%(2*k-1)==0) ans=n/(2*k-1)*2;
            else {
                ans=(n/(2*k-1))*2;
                if (n%(2*k-1)>k) ans+=2;
                else ans++;
            }
            printf("%lld\n",ans);
            ans=0;
        }
    }
    fclose(stdin);
    fclose(stdout);
    return 0;
}

T3T3T3：

题目

m=1m=1m=1 的部分分很简单。

于是打了 202020 分的部分分。

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll n,m,q;
ll a[2000005];
ll b;
ll l,r;
ll ans=0;
int main() {
    freopen("sequence.in","r",stdin);
    freopen("sequence.out","w",stdout);
    scanf("%lld %lld %lld",&n,&m,&q);
    if (m==1) {
        for (int i=1;i<=n;i++) scanf("%lld",&a[i]);
        scanf("%lld",&b);
        while(q--) {
            scanf("%lld %lld",&l,&r);
            for (int j=l;j<=r;j++) {
                if (a[j]==b) ans++;
            }
            if (ans==(r-l+1)) printf("Yes\n");
            else printf("No\n");
            ans=0;
        }
    }
    else printf("Yes\nYes\nNo"); 
    fclose(stdin);
    fclose(stdout);
    return 0;
}
